# OpenMotion SDK runtime dependencies.
#
# Core deps are also declared in pyproject.toml; this file additionally covers
# the GUI / 3D-visualization stack used by scripts/ (the live viewers and the
# perfusion head viewer), which `pip install -e .` does NOT pull in.
#
# Versions are pinned ONLY where a real constraint exists — everything else
# floats to the latest compatible release.

requests
keyboard             # optional: some scripts use hotkeys

PyQt6
